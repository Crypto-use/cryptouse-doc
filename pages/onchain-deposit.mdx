### TypeScript (Axios)

```typescript
type DepositArgs = {
  value: number;
  asset: 'USDT' | 'USDC' | 'DAI' | 'BUSD';
  network: 'eth' | 'bsc' | 'polygon';

type DepositResultSuccess = {

    paymentLinkId: string;
    network: string;

    amount: number;
  };

  message: string;
  supportedNetworks?: string[];
};

const criarDepositoOnchain = async (args: DepositArgs): Promise<DepositResult> => {
  try {
    const response = await axios.post(
      'https://api.cryptouse.com.br/api/v1/onchain/deposit',
      args,
      {
        headers: {
          'Content-Type': 'application/json',
          'Authorization': 'Bearer SEU_APP_ID'
        }
      }
    );
    
    return {
      success: true,
      order: response.data.order
    };
  } catch (error) {
    if (axios.isAxiosError(error)) {
      return {
        success: false,
        message: error.response?.data?.message || 'Ocorreu um erro durante o processo de dep√≥sito',
        supportedNetworks: error.response?.data?.supportedNetworks
      };
    }
    return {
      success: false,
      message: 'Ocorreu um erro desconhecido'
    };
  }
};

// Exemplo de uso
const executarDeposito = async () => {
  const resultado = await criarDepositoOnchain({
    value: 200,
    asset: "USDT",
    network: "bsc"
  });
  
  if (resultado.success) {
    console.log('Dep√≥sito criado:', resultado.order);
    console.log('Endere√ßo de carteira para dep√≥sito:', resultado.order.address);

    console.error('Falha no dep√≥sito:', resultado.message);
    if (resultado.supportedNetworks) {
      console.error('Redes suportadas:', resultado.supportedNetworks.join(', '));
    }
  }
};
```
